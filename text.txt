    <script>
        // Tanggal Acara yang Ditetapkan (4 Jun 2026, bermula 2:00 PM)
        const targetDate = new Date('2026-06-04T14:00:00');

        // Fungsi Timer Mundur
        function updateCountdown() {
            const now = new Date().getTime();
            const difference = targetDate - now;

            if (difference < 0) {
                document.getElementById("countdown").innerHTML = "<div class='col-span-4 text-emerald-800 font-semibold py-2'>Acara sedang berlangsung atau telah selesai</div>";
                return;
            }

            const days = Math.floor(difference / (1000 * 60 * 60 * 24));
            const hours = Math.floor((difference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((difference % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((difference % (1000 * 60)) / 1000);

            document.getElementById("days").textContent = days.toString().padStart(2, '0');
            document.getElementById("hours").textContent = hours.toString().padStart(2, '0');
            document.getElementById("minutes").textContent = minutes.toString().padStart(2, '0');
            document.getElementById("seconds").textContent = seconds.toString().padStart(2, '0');
        }

        // Jalankan timer setiap detik
        setInterval(updateCountdown, 1000);
        updateCountdown();

        // Sistem Toast Notifikasi Modern
        function tampilkanToast(pesan, jenis = "warning") {
            const toast = document.getElementById("toast");
            const toastText = document.getElementById("toast-text");
            const toastIcon = document.getElementById("toast-icon");

            toastText.textContent = pesan;

            if (jenis === "success") {
                toast.querySelector("div").className = "bg-white border-l-4 border-emerald-500 shadow-xl rounded-lg p-4 flex items-center gap-3 max-w-sm";
                toastIcon.innerHTML = '<i class="fa-solid fa-circle-check text-lg text-emerald-500"></i>';
            } else {
                toast.querySelector("div").className = "bg-white border-l-4 border-amber-500 shadow-xl rounded-lg p-4 flex items-center gap-3 max-w-sm";
                toastIcon.innerHTML = '<i class="fa-solid fa-circle-exclamation text-lg text-amber-500"></i>';
            }

            // Tampilkan animasi masuk
            toast.classList.remove("translate-y-[-100px]", "opacity-0");
            toast.classList.add("translate-y-0", "opacity-100");

            // Hilangkan otomatis setelah 3 detik
            setTimeout(() => {
                toast.classList.add("translate-y-[-100px]", "opacity-0");
                toast.classList.remove("translate-y-0", "opacity-100");
            }, 3000);
        }

        // Update Total Kehadiran
        function updateTotal() {
            const total = document.getElementById("senarai").children.length;
            document.getElementById("total-kehadiran").textContent = total;
        }

        // Fungsi Tambah Kehadiran (RSVP)
        function hadir() {
            const namaInput = document.getElementById("nama");
            const nama = namaInput.value.trim();

            if (nama === "") {
                tampilkanToast("Silakan masukkan nama Anda terlebih dahulu.");
                return;
            }

            // Tambahkan elemen daftar dengan animasi masuk yang mulus
            const li = document.createElement("li");
            li.className = "flex items-center gap-2 text-sm text-gray-700 bg-white px-3 py-2 rounded-lg border border-gray-100 transform scale-95 opacity-0 transition-all duration-300";
            li.innerHTML = `<span class="w-2 h-2 rounded-full bg-emerald-500 animate-ping"></span> ${nama}`;

            const senarai = document.getElementById("senarai");
            senarai.insertBefore(li, senarai.firstChild);

            // Trigger transisi masuk
            setTimeout(() => {
                li.classList.remove("scale-95", "opacity-0");
                li.classList.add("scale-100", "opacity-100");
            }, 50);

            // Bersihkan kolom input
            namaInput.value = "";
            
            // Perbarui jumlah total kehadiran
            updateTotal();

            // Tampilkan notifikasi berhasil
            tampilkanToast("Terima kasih! Kehadiran Anda telah berhasil dikonfirmasi.", "success");
        }
    </script>
